dependencies {
    implementation 'org.polaris2023:mps-rigid-body:0.1.4'
}

public class PhysicsMod {
    private long world;
    public void init() {
        world = RigidBodyNative.worldCreate(0, -9.81, 0);
        RigidBodyNative.worldRegisterCelestialGravity(world, 3, 8);
        ByteBuffer arena = RigidBodyNative.worldGetArenaDirectByteBuffer(world);
    }
    public void onServerTick() {
        RigidBodyNative.worldStep(world, 1.0 / 20.0);
    }
    public void shutdown() {
        RigidBodyNative.worldDestroy(world);
    }
}
